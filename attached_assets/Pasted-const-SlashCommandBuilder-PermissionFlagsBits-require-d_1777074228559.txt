const { SlashCommandBuilder, PermissionFlagsBits } = require('discord.js');

module.exports = {
  data: new SlashCommandBuilder()
    .setName('lottery')
    .setDescription('Run a draft lottery')
    .setDefaultMemberPermissions(PermissionFlagsBits.Administrator) // 🔒 Admin only
    .addIntegerOption(option =>
      option.setName('count')
        .setDescription('Number of participants (2-32)')
        .setRequired(true)
    ),

  async execute(interaction) {
    const APPROVED_ROLE_ID = 'YOUR_ROLE_ID_HERE';
    let count = interaction.options.getInteger('count');

    if (count < 2 || count > 32) {
      return interaction.reply({ content: 'Choose between 2 and 32 participants.', ephemeral: true });
    }

    await interaction.deferReply();

    const guild = interaction.guild;
    const members = await guild.members.fetch();

    // Filter approved members
    const approvedMembers = members.filter(m =>
      m.roles.cache.has(APPROVED_ROLE_ID) && !m.user.bot
    );

    if (approvedMembers.size < 2) {
      return interaction.editReply('Not enough approved members.');
    }

    let pool = Array.from(approvedMembers.values());

    // Shuffle full pool
    pool = shuffleArray(pool);

    // Select requested count
    const selected = pool.slice(0, count);

    // Final randomized draft order
    const results = shuffleArray([...selected]);

    // 🎰 Pool Preview
    let preview = `🎰 **DRAFT LOTTERY** 🎰\n\n`;
    preview += `**Participants (${selected.length}):**\n`;
    preview += selected.map(m => `• ${m.user.username}`).join('\n');

    const message = await interaction.editReply(preview);

    const wait = ms => new Promise(res => setTimeout(res, ms));

    // 🥁 Drumroll
    await wait(1500);
    await message.edit(`🥁 **DRUMROLL PLEASE...** 🥁`);
    await wait(2000);

    let revealed = [];

    for (let i = 0; i < results.length; i++) {

      // 🎲 Rolling animation (3–4 spins, slightly randomized speed)
      const spins = 3 + Math.floor(Math.random() * 2);

      for (let spin = 0; spin < spins; spin++) {
        const fake = results[Math.floor(Math.random() * results.length)];

        let rollingDisplay = `🎰 **Drawing Pick #${i + 1}...**\n\n`;
        rollingDisplay += revealed.join('\n') + '\n\n';
        rollingDisplay += `🎲 Rolling... **${fake.user.username}**`;

        await message.edit(rollingDisplay);

        // Variable delay for realism
        await wait(400 + Math.random() * 300);
      }

      // 🏆 Real reveal
      const user = results[i];
      const pickText = formatPick(i + 1);

      revealed.push(`**${pickText}:** <@${user.id}>`);

      let revealDisplay = `🎰 **Draft Lottery Results**\n\n`;
      revealDisplay += revealed.join('\n');

      await message.edit(revealDisplay);

      // Slightly increasing delay as picks go on
      await wait(900 + i * 150);
    }

    // 🏁 Final Board
    let finalDisplay = `🏆 **FINAL DRAFT ORDER** 🏆\n\n`;
    finalDisplay += revealed.join('\n');

    await message.edit(finalDisplay);
  }
};

// 🎲 Shuffle
function shuffleArray(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
  return array;
}

// 🥇 Pick labels
function formatPick(num) {
  if (num === 1) return '1st Pick';
  if (num === 2) return '2nd Pick';
  if (num === 3) return '3rd Pick';
  return `${num}th Pick`;
}
