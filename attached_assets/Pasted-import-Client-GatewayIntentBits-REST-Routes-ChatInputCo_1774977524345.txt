import { Client, GatewayIntentBits, REST, Routes, ChatInputCommandInteraction } from "discord.js";
import axios from "axios";
import fs from "fs";
import path from "path";
import AdmZip from "adm-zip";
import dotenv from "dotenv";
import { drizzle } from "drizzle-orm/node-postgres";
import pkg from "pg";
import { pgTable, text, integer } from "drizzle-orm/pg-core";
import { eq } from "drizzle-orm";

dotenv.config();

const { Pool } = pkg;

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
});

const db = drizzle(pool);

const teamsTable = pgTable("teams", {
  id: text("id").primaryKey(),
  name: text("name"),
  user: text("user"),
  wins: integer("wins"),
  losses: integer("losses"),
  pf: integer("pf"),
  pa: integer("pa"),
  coins: integer("coins"),
});

const processedGames = pgTable("processed_games", {
  gameId: text("game_id").primaryKey(),
});

const rankingsTable = pgTable("rankings", {
  team: text("team").primaryKey(),
  rank: integer("rank"),
});

const DATA_DIR = "./data/raw";

function loadJSON(file: string) {
  return JSON.parse(fs.readFileSync(path.join(DATA_DIR, file), "utf-8"));
}

async function parseFranchise() {
  const tData = loadJSON("teams.json");
  const roster = loadJSON("roster.json");
  const stats = loadJSON("stats.json");
  const schedules = loadJSON("schedules.json");

  const teamMap: Record<string, any> = {};
  const playerMap: Record<string, any> = {};

  Object.values(tData).forEach((t: any) => {
    const id = t.teamId ?? t.teamIndex;
    teamMap[id] = {
      id,
      name: `${t.cityName} ${t.teamName}`,
      user: t.userName || "CPU",
      wins: 0,
      losses: 0,
      pf: 0,
      pa: 0,
      coins: 0,
    };
  });

  roster.forEach((p: any) => {
    playerMap[p.rosterId] = {
      name: `${p.firstName} ${p.lastName}`,
      teamId: p.teamIndex,
    };
  });

  const season = Object.values(schedules.reg)[0] as any;

  for (const week of Object.values(season)) {
    for (const game of Object.values(week as any)) {
      if (!game.homeTeamId || !game.awayTeamId) continue;
      if (game.status !== 3) continue;

      const gameId =
        game.gameId ||
        `${game.weekIndex}-${game.homeTeamId}-${game.awayTeamId}`;

      const exists = await db
        .select()
        .from(processedGames)
        .where(eq(processedGames.gameId, gameId));

      if (exists.length > 0) continue;

      const home = teamMap[game.homeTeamId];
      const away = teamMap[game.awayTeamId];

      home.pf += game.homeScore;
      home.pa += game.awayScore;
      away.pf += game.awayScore;
      away.pa += game.homeScore;

      if (game.homeScore > game.awayScore) {
        home.wins++;
        away.losses++;
      } else {
        away.wins++;
        home.losses++;
      }

      const isUserGame =
        home.user !== "CPU" && away.user !== "CPU";

      if (isUserGame) {
        if (game.homeScore > game.awayScore) {
          home.coins += 50;
          away.coins += 20;
        } else {
          away.coins += 50;
          home.coins += 20;
        }
      } else {
        const userTeam = home.user !== "CPU" ? home : away;
        const win =
          (home.user !== "CPU" && game.homeScore > game.awayScore) ||
          (away.user !== "CPU" && game.awayScore > game.homeScore);

        if (win) userTeam.coins += 20;
      }

      await db.insert(processedGames).values({ gameId });
    }
  }

  for (const t of Object.values(teamMap)) {
    await db.insert(teamsTable).values(t as any).onConflictDoUpdate({
      target: teamsTable.id,
      set: t as any,
    });
  }

  return Object.values(teamMap);
}

async function generatePowerRankings(teamList: any[]) {
  const prev = await db.select().from(rankingsTable);

  const ranked = teamList
    .filter((t) => t.user !== "CPU")
    .map((t) => {
      const diff = t.pf - t.pa;
      const score = t.wins * 10 + diff * 0.1;
      return { ...t, diff, score };
    })
    .sort((a, b) => b.score - a.score);

  const output: string[] = [];

  for (let i = 0; i < ranked.length; i++) {
    const team = ranked[i];
    const prevRank = prev.find((p) => p.team === team.name)?.rank;

    let movement = "";
    if (prevRank !== undefined) {
      const change = prevRank - (i + 1);
      if (change > 0) movement = `(+${change}) `;
      if (change < 0) movement = `(${change}) `;
    }

    output.push(
      `${movement}**${i + 1}. ${team.name}** ${team.wins}-${team.losses} | PD: **${team.diff}**`
    );

    await db.insert(rankingsTable).values({
      team: team.name,
      rank: i + 1,
    }).onConflictDoUpdate({
      target: rankingsTable.team,
      set: { rank: i + 1 },
    });
  }

  return output;
}

async function handleUpdate(interaction: ChatInputCommandInteraction) {
  const attachment = interaction.options.getAttachment("file");

  if (!attachment || !attachment.name.endsWith(".zip")) {
    return interaction.reply({
      content: "❌ Upload a ZIP file.",
      ephemeral: true,
    });
  }

  await interaction.reply("📥 Processing...");

  const zipPath = "./data/upload.zip";

  const res = await axios({
    url: attachment.url,
    method: "GET",
    responseType: "stream",
  });

  const writer = fs.createWriteStream(zipPath);
  res.data.pipe(writer);

  await new Promise((r) => writer.on("finish", r));

  fs.rmSync(DATA_DIR, { recursive: true, force: true });
  fs.mkdirSync(DATA_DIR, { recursive: true });

  const zip = new AdmZip(zipPath);
  zip.extractAllTo(DATA_DIR, true);

  const teams = await parseFranchise();
  const rankings = await generatePowerRankings(teams);

  const payouts = teams
    .filter((t: any) => t.user !== "CPU" && t.coins > 0)
    .map((t: any) => `💰 **${t.name}** (+${t.coins})`);

  await interaction.followUp("✅ Update complete.");

  if (payouts.length) {
    await interaction.channel?.send(
      `**💰 Coin Payouts**\n${payouts.join("\n")}`
    );
  }

  await interaction.channel?.send(
    `**🏆 H2H POWER RANKINGS**\n@everyone\n\n${rankings.join("\n")}`
  );
}

const client = new Client({
  intents: [GatewayIntentBits.Guilds],
});

client.once("ready", () => {
  console.log(`✅ Logged in as ${client.user?.tag}`);
});

client.on("interactionCreate", async (interaction) => {
  if (!interaction.isChatInputCommand()) return;

  if (interaction.commandName === "update") {
    await handleUpdate(interaction);
  }
});
